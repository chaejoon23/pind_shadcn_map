### WSL 사용하지 않는 경우, WSL 삭제하여 컴퓨터 리소스 확보하기 ###
(파워셀에서 다음 명령어 실행하기)
wsl --uninstall

(파워셸에서 다음 명령어 붙여 넣고 엔터를 누르기)

Get-ChildItem HKCU:\SOFTWARE\Microsoft\Windows\CurrentVersion\Lxss |
  ForEach-Object {
      $p = Get-ItemProperty $_.PsPath
      [PSCustomObject]@{
          Name     = $p.DistributionName
          BasePath = $p.BasePath
      }
  }
